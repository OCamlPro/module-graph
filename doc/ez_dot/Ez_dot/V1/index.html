Module Ez_dot.V1

type graph = {
  1. graph_name : string;
  2. mutable graph_nodes : node list;
  3. mutable graph_edges : edge list;
  4. mutable node_counter : int;
  5. graph_attributes : graph_attributes list;
}
and node = {
  1. mutable node_name : string;
  2. node_id : int;
  3. node_graph : graph;
  4. node_attributes : node_attributes list;
}
and edge = {
  1. edge_from : node;
  2. edge_to : node;
  3. edge_attributes : edge_attributes list;
}
and graph_attributes =
  1. | GraphSize of float * float
and node_attributes =
  1. | NodeColor of string
  2. | NodeFontColor of string
  3. | NodeFontName of string
  4. | NodeShape of shape
  5. | NodeHeight of float
  6. | NodeWidth of float
  7. | NodeStyle of goptions
and edge_attributes =
  1. | EdgeDirection of direction
  2. | EdgeLabel of string
  3. | EdgeStyle of goptions
  4. | EdgeWeight of int
and direction =
  1. | Forward
  2. | Backward
  3. | Bothdir
  4. | Nodir
and goptions =
  1. | Bold
  2. | Dotted
  3. | Filled
and shape =
  1. | Ellipse
  2. | Box
  3. | Circle
  4. | DoubleCircle
  5. | Diamond
  6. | PlainText
  7. | Record
  8. | Polygon of int * polygon_options list
  9. | Epsf of string
and polygon_options =
  1. | Skew of float
  2. | Distortion of float
val create : string -> graph_attributes list -> graph
val node : graph -> string -> node_attributes list -> node
val edge : node -> node -> edge_attributes list -> edge
val add_edge : node -> node -> edge_attributes list -> unit
val add_edges : node -> node list -> edge_attributes list -> unit
val add_path : node list -> edge_attributes list -> unit
val rename_node : node -> string -> unit
val save : graph -> string -> unit
val save_in : graph -> Stdlib.out_channel -> unit
val view : graph -> unit
val dot2file : dotfile:string -> format:string -> outfile:string -> unit
